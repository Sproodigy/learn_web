normalize()

Description:
The normalize() method removes empty Text nodes, and joins adjacent Text nodes.

Examples

The paragraph has child node(s).

<body onload="normPara()">
<div>

<p> The paragraph has <span class="replace"> <span> child node(s). </p>

</div>
</body>
function addTextNode() {

var x = element.createTextNode("Click again.");
var y = element.getElementsByTagName("p")[0];
y.appendChild(x);
var x = element.getElementsByClassName("replace")[0];
x.innerHTML = y.childNodes.length;

}
function normPara() {

element.normalize();
var y = element.getElementsByTagName("p")[0];
var x = element.getElementsByClassName("replace")[0];
x.innerHTML = y.childNodes.length;

}
