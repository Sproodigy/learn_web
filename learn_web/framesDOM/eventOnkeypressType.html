onkeypress

Description:
The onkeypress event occurs when the user presses a key (on the keyboard).
The onkeypress event is not fired for all keys (e.g. ALT, CTRL, SHIFT, ESC) in all browsers. To detect only whether the user has pressed a key, use the onkeydown event instead, because it works for all keys.
Supported HTML tags:
All HTML elements, EXCEPT: <base>, <bdo>, <br>, <head>, <html>, <iframe>, <meta>, <param>, <script>, <style>, and <title>.

Examples

Onkeydown: Onkeypress: Onkeyup:

<body onkeydown="eventOnkeydownFunc(event)" onkeypress="eventOnkeypressFunc(event)" onkeyup="eventOnkeyupFunc(event)">

<span class="replace"> </span>
<span class="replace"> </span>
<span class="replace"> </span>

</body>

function eventOnkeydownFunc(event) {

var k = event.key;
var c = event.code;
var kc = event.keyCode;
var cc = event.charCode;
var w = event.which;
var text = "key: " + k + "code: " + c + "keyCode: " + kc + "charCode: " + cc + "which: " + w;
var x = document.getElementsByClassName("replace")[0];
x.innerHTML = text;

}
