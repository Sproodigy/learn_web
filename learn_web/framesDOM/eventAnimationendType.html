animationend

Description:
The animationend event occurs when a CSS animation has completed.
Notes:
For Chrome, Safari and Opera, use the webkitAnimationEnd prefix.

Examples

<body>
<div id="eventAnimationLoop"> </div>
</body>
<style>
#eventAnimationLoop {

width: 350px;
height: 135px;
background-color: green;
position: relative;

}
</style>
<script>

var x = document.getElementById("eventAnimationLoop");

function startAnimationFunc() {

x.style.WebkitAnimation = "someAnimation 3s 2";
x.style.animation = "someAnimation 3s 2";

}

x.addEventListener("webkitAnimationStart", animationStartFunc);
x.addEventListener("webkitAnimationIteration", animationIterationFunc);
x.addEventListener("webkitAnimationEnd", animationEndFunc);

x.addEventListener("animationstart", animationStartFunc);
x.addEventListener("animationiteration", animationIterationFunc);
x.addEventListener("animationend", animationEndFunc);

function animationStartFunc() {

this.innerHTML += "The animation has started!<br>";

}

function animationIterationFunc() {

this.innerHTML += "Animation repeats!<br>";

}

function animationEndFunc() {

this.innerHTML += "The animation has completed!";

}
</script>
