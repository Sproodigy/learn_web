position

Description:
The position property returns the position and altitude of the device on Earth.

Coordinates Properties

PropertyDescription
position.coords Returns a Coordinates object that defines the current location.
position.timestamp Returns a DOMTimeStamp representing the time at which the location was retrieved.

Examples

<body>
<div onclick="eventGeolocationFunc()">

<p class="replace"> </p>

</div>
</body>
var x = document.getElementsByClassName("replace")[0];

function eventGeolocationFunc() {

if (navigator.geolocation) {
x.innerHTML = navigator.geolocation.getCurrentPosition(showPosition);
} else {
x.innerHTML = "Geolocation is not supported by this browser.";
}

}

function showPosition(pos) {

x.innerHTML = "Latitude: " + pos.coords.latitude + "<br>Longitude: " + pos.coords.longitude + "<br>Altitude: " + pos.coords.altitude + "<br>Accuracy: " + pos.coords.accuracy + "<br>Altitude accuracy: " + pos.coords.altitudeAccuracy + "<br>Heading: " + pos.coords.heading + "<br>Speed: " + pos.coords.speed + "<br>Time: " + pos.timestamp;

}
