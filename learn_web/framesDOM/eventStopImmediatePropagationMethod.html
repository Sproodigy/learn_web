stopImmediatePropagation()

Description:
The stopImmediatePropagation() method prevents other listeners of the same event from being called.

Examples

Cancel immediate propagation:
<div>

<button>Perform</button>
<p class="example"> </p>

</div>

Cancel immediate propagation:
<input type="checkbox" id="check">
function printText(event) {

var x = document.getElementsByClassName("replace")[0];
x.innerHTML = "Example text...";

if (document.getElementById("check").checked) {
event.stopImmediatePropagation();
}

}

function addText() {

document.getElementsByClassName("replace")[0];
x.innerHTML += " other text...";

}

document.getElementsByTagName("button")[0];
x.addEventListener("click", printText);
x.addEventListener("click", addText);
